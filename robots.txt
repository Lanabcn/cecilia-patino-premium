const initBA = () => {
  document.querySelectorAll('.ba').forEach((ba)=>{
    const before = ba.querySelector('img.before');
    const after  = ba.querySelector('img.after');
    const range  = ba.querySelector('input[type="range"]');
    let handle = ba.querySelector('.handle');
    if(!handle){ handle = document.createElement('div'); handle.className = 'handle'; ba.appendChild(handle); }
    let knob = ba.querySelector('.knob');
    if(!knob){ knob = document.createElement('div'); knob.className = 'knob'; ba.appendChild(knob); }

    const setPosition = (value)=>{
      const pct = Number(value);
      if(after){ after.style.clipPath = `inset(0 0 0 ${pct}%)`; }
      if(handle){ handle.style.left = pct + '%'; }
      if(knob){ knob.style.left = pct + '%'; }
    };

    const start = ba.dataset.initial || 50;
    if(range){ range.value = start; setPosition(start); }

    if(range){
      range.addEventListener('input', (e)=> setPosition(e.target.value));
      const updateFromEvent = (clientX)=>{
        const rect = ba.getBoundingClientRect();
        let pct = (clientX - rect.left) / rect.width * 100;
        pct = Math.max(0, Math.min(100, pct));
        range.value = pct.toFixed(2);
        setPosition(pct);
      };
      ba.addEventListener('pointerdown', (e)=>{ ba.setPointerCapture(e.pointerId); updateFromEvent(e.clientX); });
      ba.addEventListener('pointermove',  (e)=>{ if(e.pressure>0 || e.buttons===1){ updateFromEvent(e.clientX); } });
      range.addEventListener('keydown', (e)=>{
        const step = (e.shiftKey? 10 : 1);
        let v = Number(range.value);
        if(e.key==='ArrowLeft'){ v-=step; }
        if(e.key==='ArrowRight'){ v+=step; }
        if(e.key==='Home'){ v=0; }
        if(e.key==='End'){ v=100; }
        v=Math.max(0,Math.min(100,v));
        range.value=v; setPosition(v);
      });
    }

    // Alt desde data-attrs
    if(ba.dataset.altBefore && before){ before.alt = ba.dataset.altBefore; }
    if(ba.dataset.altAfter  && after){  after.alt  = ba.dataset.altAfter;  }
  });
};

window.addEventListener('DOMContentLoaded', initBA);
